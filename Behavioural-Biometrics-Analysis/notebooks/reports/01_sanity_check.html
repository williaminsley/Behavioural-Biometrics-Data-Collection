def find_data_dir(start: Path) -> Path:
    start = start.resolve()
    for p in [start] + list(start.parents):
        candidate = p / "data" / "raw" / "sessions"
        if candidate.exists():
            return candidate
    raise FileNotFoundError(
        "Could not find data/raw/sessions by walking up from: " + str(start)
    )

checks = []
